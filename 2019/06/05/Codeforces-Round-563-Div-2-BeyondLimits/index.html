C.Ehab and a Special Coloring Problem

质数显然互质，故不可相同。

include

static const int MAXN=100050; using namespace std; int n,cnt,num[MAXN]; int main(){
cin>>n; for(int i=2;i<=n;i++){ if(!num[i]){
num[i]\=++cnt; for(int j=i;j<=n;j+=i) num[j]=cnt;
}
printf(“%d “,num[i]);
} return 0;
}
